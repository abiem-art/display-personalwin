let current = 1; const totalPages = 6; const duration = 10000; // Durasi 10 detik (dalam milidetik) setInterval(() => { // Targetkan halaman saat ini dan halaman berikutnya const oldPage = document.getElementById('page' + current); current = (current % totalPages) + 1; const nextPage = document.getElementById('page' + current); // Lakukan perpindahan class active oldPage.classList.remove('active'); nextPage.classList.add('active'); // Refresh halaman yang lama di latar belakang agar data tetap update saat muncul lagi setTimeout(() => { oldPage.src = oldPage.src; }, 2000); }, duration);
